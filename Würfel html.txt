    <script>
        function roll() {
            const dice = document.getElementById('dice');
            const message = document.getElementById('message');
            
            // Animation hinzufügen
            dice.classList.add('shake');
            dice.innerText = "...";
            message.innerText = "Der Würfel rollt...";

            setTimeout(() => {
                const zahl = Math.floor(Math.random() * 6) + 1;
                dice.innerText = zahl;
                dice.classList.remove('shake');
                
                // Kurze motivierende Texte je nach Zahl
                if(zahl === 6) {
                    message.innerText = "EINE SECHS! Dein Zug!";
                } else {
                    message.innerText = "Du hast eine " + zahl + " gewürfelt.";
                }
            }, 400);
        }
    </script>
